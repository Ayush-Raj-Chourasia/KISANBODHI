   ```typescript
   const { data } = useAdvisory(farmer, true)

     return format(data.sentinelReport.marketData)
   }
   ```
3. **React Query Hook** (`useAdvisory`):

   ```typescript
   fetch('http://localhost:3001/api/advisory', {

     timeout: 30000,
     signal: AbortSignal
   })
   ```

   - Returns mock market data immediately
   - Includes crops: wheat, rice, sugarcane, cotton (with trends)
   - Includes prices: ₹2,400-3,200 per quintal

## 🎓 Developer Reference

### Add New API Endpoint (Easy 3-Step Process)

**Step 1**: Add to `src/services/api.client.ts`
```typescript
async getCropYield(farmer: Farmer): Promise<YieldResponse> {
  return this.request<YieldResponse>('/crop-yield', {
    method: 'POST',
    body: JSON.stringify({ farmer })
  })
}
```

**Step 2**: Create hook in `src/hooks/useApi.ts`
```typescript
export const useYield = (farmer?: Farmer, enabled = true) => {
  return useQuery({
    queryKey: [...queryKeys, 'yield', farmer?.id],
    queryFn: () => apiClient.getCropYield(farmer!),
    staleTime: 5 * 60 * 1000, // 5 minutes
    enabled: enabled && !!farmer
  })
}
```

**Step 3**: Use in component
```typescript
const { data: yieldData } = useYield(farmer)
return <div>{yieldData?.estimatedYield} quintals/hectare</div>
```

### Backend-Frontend Type Sync

Types are defined in `src/services/api.client.ts` and shared:
```typescript
// Frontend import
import { Alert, Farmer, AdvisoryResponse } from '../services/api.client'

// These match backend types exactly
// If backend adds a field, TypeScript will error until frontend updates
```

## 🚨 Troubleshooting

| Issue | Solution |
|-------|----------|
| "Cannot connect to backend" | Backend not running? Run `cd backend && npm run dev` |
| "Loading spinner never stops" | Check backend logs for errors or CORS issues |
